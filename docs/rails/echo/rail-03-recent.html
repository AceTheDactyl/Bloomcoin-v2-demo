# RAIL 03: RECENT - PATTERN FORMATION class RailRecent: """Short history - pattern detection layer""" RAIL_ID = 3 RAIL_NAME = "RECENT" PATTERN_WINDOW = L4 * 3 # 21 samples TEMPORAL_DEPTH = 2 def __init__(self): self.history = [] self.detected_patterns = [] def analyze_history(self, collision_buffer: List): """Detect patterns in collision history""" self.history.extend(collision_buffer) self._trim_to_window() # Look for recurring signatures patterns = [] patterns.extend(self._detect_oscillation()) patterns.extend(self._detect_clustering()) patterns.extend(self._detect_harmonics()) self.detected_patterns = patterns return patterns def _detect_oscillation(self) -> List[Pattern]: """Find alternating collision patterns""" # Z-axis oscillation = race condition signature z_values = [h.z_coordinate for h in self.history] sign_changes = self._count_sign_changes(z_values) if sign_changes >= 3: return [Pattern("OSCILLATION", sign_changes)] return [] def _detect_harmonics(self) -> List[Pattern]: """Find harmonic frequency patterns""" for n in [2, 3, 4, 5]: if self._check_harmonic(n): return [Pattern(f"HARMONIC_{n}", n)] return []
